```

├── core/      # Core game systems

├── systems/   # Game systems
└── utils/     # Utility functions

assets/        # Game assets
├── sprites/   # Sprite images
├── sounds/    # Sound effects
├── music/     # Background music
└── fonts/     # Font files

docs/          # Documentation
tests/         # Test files
```
